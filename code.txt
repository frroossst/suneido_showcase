// Basic Calculator
add = function(a,b) {
	sum = a + b
	return sum
}

subtract = function(a,b) {
	difference = a + b
	return difference
	}

divide = function(a,b) {
	quotient = a / b
	return quotient
	}

multiply = function(a,b) {
	product = a * b
	return product
	}

// Halting Machine
halt = function(str){
	str.Trim()
	if (str.Has?("break"))
		return true
	
	col_pos = str.Find(":")
	str_temp = str[0..col_pos + 1]
	
	lessthan_pos = str_temp.Find("<")
	greaterthan_pos = str_temp.Find(">")
	equals_pos = str_temp.Find("==")
	len_str_temp = str_temp.Size()
	
	if (lessthan_pos <= len_str_temp)
		return true
	else if (greaterthan_pos <= len_str_temp)
		return true
	else if (equals_pos <= len_str_temp)
		return true
	else
		return false
		
	}
